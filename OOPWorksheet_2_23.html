Quick Reference — Key Ideas
A class is a blueprint. An object is a thing built from that blueprint using new.
An attribute is data the object stores. A method is something the object can do.
We call a method like this:  objectName.methodName()
ArrayList — a list that can grow:  list.add(x)  |  list.get(i)  |  list.size()

// ----- Student.java -----
public class Student {
    private String name;
    private int    grade;

    public Student(String name, int grade) {
        this.name  = name;
        this.grade = grade;
    }

    public String getName()  { return name; }
    public int    getGrade() { return grade; }

    public String getResult() {
        if (grade >= 75) {
            return name + " passed!";
        } else {
            return name + " needs to study more.";
        }
    }
}
// ----- Classroom.java -----
import java.util.ArrayList;

public class Classroom {
    private ArrayList<Student> students = new ArrayList<>();

    public void add(Student s) {
        students.add(s);
    }

    public void printResults() {
        int i = 0;
        while (i < students.size()) {
            Student s = students.get(i);
            System.out.println(s.getResult());
            i++;
        }
    }
}
// ----- Main.java -----
public class Main {

        Classroom room = new Classroom();
        room.add(new Student("Ana",  82));
        room.add(new Student("Ben",  65));

    }

Trace carefully — a method modifies an object that's stored inside another object.

// ----- BankAccount.java -----
public class BankAccount {
    private String owner;
    private double balance;

    public BankAccount(String owner, double startingBalance) {
        this.owner   = owner;
        this.balance = startingBalance;
    }

    public void deposit(double amount)  { balance += amount; }
    public void withdraw(double amount) { balance -= amount; }
    public double getBalance()          { return balance; }
    public String getOwner()            { return owner; }
}
// ----- Bank.java -----
import java.util.ArrayList;

public class Bank {
    private ArrayList<BankAccount> accounts = new ArrayList<>();

    public void openAccount(BankAccount acc) {
        accounts.add(acc);
    }

    public void printSummary() {
        int i = 0;
        while (i < accounts.size()) {
            BankAccount acc = accounts.get(i);
            System.out.println(acc.getOwner() + ": $" + acc.getBalance());
            i++;
        }
    }
}
// ----- Main.java -----
public class Main {
    public static void main(String[] args) {
        BankAccount alice = new BankAccount("Alice", 500.0);
        BankAccount bob   = new BankAccount("Bob",   200.0);

        alice.deposit(150.0);
        bob.withdraw(50.0);
        alice.withdraw(75.0);

        Bank myBank = new Bank();
        myBank.openAccount(alice);
        myBank.openAccount(bob);
        myBank.printSummary();
    }
}
